    function calculateProfit() {
      const cost = parseFloat(document.getElementById("costPrice").value);
      const sell = parseFloat(document.getElementById("sellingPrice").value);
      const resultDiv = document.getElementById("result");

      if (isNaN(cost) || isNaN(sell)) {
        resultDiv.innerHTML = "‚ùå Please enter valid numbers.";
        return;
      }

      const profit = sell - cost;
      const margin = ((profit / cost) * 100).toFixed(2);

      resultDiv.innerHTML = `‚úÖ Profit: ‚Ç¶${profit.toLocaleString()} <br />üíπ Profit Margin: ${margin}%`;
    }
